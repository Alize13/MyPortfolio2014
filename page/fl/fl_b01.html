彈跳

 

作品展示

程式碼

var mvy=5;
var g=2;//重力加速度

ball_mc.addEventListener(Event.ENTER_FRAME, mv);
function mv(event:Event):void{
	mvy+=g;//越彈越快
	if(ball_mc.y>350){
		ball_mc.y=350;//不讓他跑到地板下
		mvy=-mvy;//讓mvy變負數反彈
		mvy+=3;//反彈的同時讓他反彈力道變小(一定要比g大效果才會出來)
	}
	ball_mc.y+=mvy; 
}

 

 

 

控制元件的各種屬性:方向.大小.旋轉角度.透明度

 

作品展示

程式碼

import flash.events.MouseEvent

//往上飛、y軸遞減
up_btn.addEventListener(MouseEvent.CLICK, up);
function up(event:MouseEvent):void
{
	//bird_mc.y = bird_mc.y - 5;
	bird_mc.y-=5;
}

//往下飛、y軸遞增
down_btn.addEventListener(MouseEvent.CLICK, down);

function down(event:MouseEvent):void
{
	bird_mc.y+=5;
}

//往左飛、x軸遞減
left_btn.addEventListener(MouseEvent.CLICK, left);

function left(event:MouseEvent):void
{
	bird_mc.x-=5;
}

//往右飛、x軸遞增
right_btn.addEventListener(MouseEvent.CLICK, right);

function right(event:MouseEvent):void
{
	bird_mc.x+=5;
}


//放大、scaleX & scaleX縮放倍率 倍率建議0~1
big_btn.addEventListener(MouseEvent.CLICK, big);

function big(event:MouseEvent):void
{
	bird_mc.scaleX+=0.1;
	bird_mc.scaleY=bird_mc.scaleX;
}

//縮小、scaleX & scaleX縮放倍率 倍率建議0~1
small_btn.addEventListener(MouseEvent.CLICK, small);

function small(event:MouseEvent):void
{
	if(bird_mc.scaleX>0.1){
		bird_mc.scaleX-=0.1;
		bird_mc.scaleY-=0.1;
	}
	else{
		trace("無法再縮小了");
	}
}

//左轉、rotation旋轉角度 -為逆時針
r1_btn.addEventListener(MouseEvent.CLICK,r1);

function r1(event:MouseEvent):void
{
	bird_mc.rotation-=30;
}

//右轉、rotation旋轉角度 +為順時針
r2_btn.addEventListener(MouseEvent.CLICK,r2);

function r2(event:MouseEvent):void
{
	bird_mc.rotation+=30;
}

//透明度增加、alpha透明度百分比 倍率建議0~1
a1_btn.addEventListener(MouseEvent.CLICK,a1);

function a1(event:MouseEvent):void
{
	bird_mc.alpha-=0.1;
}

//透明度減少、alpha透明度百分比 倍率建議0~1
a2_btn.addEventListener(MouseEvent.CLICK,a2);

function a2(event:MouseEvent):void
{
	bird_mc.alpha+=0.1;
}

//隱藏、visible為false 
v1_btn.addEventListener(MouseEvent.CLICK,v1);

function v1(event:MouseEvent):void
{
	bird_mc.visible=false;
}

//顯現、visible為true 
v2_btn.addEventListener(MouseEvent.CLICK,v2);

function v2(event:MouseEvent):void
{
	bird_mc.visible=true;
}

//重設
reset_btn.addEventListener(MouseEvent.CLICK,reset1);

function reset1(event:MouseEvent):void
{
	bird_mc.visible=true;
	bird_mc.alpha=1;
	bird_mc.rotation=0;
	bird_mc.scaleX=1;
	bird_mc.scaleY=1;
	bird_mc.x=288;
	bird_mc.y=190;
}
