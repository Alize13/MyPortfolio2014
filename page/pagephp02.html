程式碼

< ?php
/*系統完整日期西元4位數(年-月-日)*/ 
echo date('Y-m-d')."< br />";
/*系統完整日期西元2位數(年/月/日)*/
echo date('y/m/d')."< br />"; 
/*系統完整日期西元4位數年月不補0(年/月/日)*/
echo date('Y/n/j')."< br />"; 
/*系統完整日期西元2位數(日/月/年)*/
echo date('d/m/y')."< br />";
/*中文混合+系統完整日期西元4位數年月不補0(年/月/日)*/
echo date('y年n月j日')."< br />"; 
/*顯示今天星期幾(英文形)中文字打外面*/
echo "今天是".date('l')."< br />";  
/*系統時間24小時制(時/分/秒)時區+0*/
echo date('H:i:s')."< br />"; 
/*系統時間24小時制(時/分/秒)台灣時區*/
date_default_timezone_set('Asia/Taipei');
echo date('H:i:s')."< br />";
/*系統時間12小時制(時/分/秒)時區+0 A為大小AM/PM a為小寫am/pm */
echo date('A h:i:s')."< br />"; 
?>
