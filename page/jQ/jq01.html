＊jQ02片段:動畫box02從透明漸漸浮現

(注意.box02與.box02 imgCSS樣式中的定位)

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

 

程式碼

	$(document).ready(function(){
    	/*修改css*/
		$('.box02 img').css(
			{'opacity':'0',/*透明度0到1之間 0為完全透明 1為不透明*/
			'width':'0%',
			'height':'0%',
			'top':'50%',
			'left':'50%' });
        /*製作動畫*/
		$('.box02 img').animate(
			{'opacity':'1',
			'width':'100%',
			'height':'100%',
			'top':'0%',
			'left':'0%'},3000)})
