JQuery easing外掛

＊jQ03片段:利用jQuery easing外掛製作動畫

(效果類似ActionScript的Tweener)

程式碼

		$(document).ready(function(){
		/*宣告一陣列儲存想要的運動方式*/
		var motion=new Array(
  'easeInElastic','easeInBounce','easeOutElastic','easeInOutBack','easeInQuart');
		/*寫一個each迴圈(類似for迴圈功能)  PS:i不用宣告就可以用喔*/
$('.box03').each(function(i){
			$(this).css('left',i*100+10*i+"px");
			$(this).animate({'top':'500px'},2500,motion[i]);
			/*append加入*/
			$(this).append(motion[i]);
		})
	})
