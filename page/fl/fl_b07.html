Tweener應用

利用Tweener做出動態的menu bar

 

作品展示

程式碼

import caurina.transitions.Tweener;
var prev = 1;
var now = 1;
var dir = 1;
var intervalId = setInterval($next,2000);
setInterval($jump,250);
function $jump()
{

	if (dir==1)
	{
		Tweener.addTween(dm, {y:340, time:0.25, transition:"easeOutCirc"});
		dir=0;
	}
	else
	{
		Tweener.addTween(dm, {y:360, time:0.25, transition:"easeInCirc"});
		dir=1;
	}

}
function $next()
{
	prev = now;
	now++;
	if (now==6)
	{
		now = 1;
	}
	$move();
}

for (var i=1; i<=5; i++)
{
	this["b" + i].addEventListener(MouseEvent.CLICK,$btnHandler);
}

function $move()
{
	if (now< prev)
	{
		dm.rotation = (now - prev) * 10 - 20;
	}
	else
	{
		dm.rotation = (now - prev) * 10 + 20;
	}
	Tweener.addTween(dm, {x:(now-1)*110+55,rotation:0, time:1, transition:"easeOutCirc"});
}


function $btnHandler($e:MouseEvent)
{
	clearInterval(intervalId);
	intervalId = setInterval($next,2000);
	switch ($e.currentTarget.name)
	{
		case "b1" :
			prev = now;
			now = 1;
			$move();
			break;
		case "b2" :
			prev = now;
			now = 2;
			$move();
			break;
		case "b3" :
			prev = now;
			now = 3;
			$move();
			break;
		case "b4" :
			prev = now;
			now = 4;
			$move();
			break;
		case "b5" :
			prev = now;
			now = 5;
			$move();
			break;

	}
}
