＊jQ05片段:連續動畫+延遲

 

 

 

 

 

程式碼

	$(document).ready(function(){    /*delay 完成此動作後停留(毫秒)才繼續下一個動作*/
		$('.box04').animate({'left':'400px'},1000,'easeOutElastic').delay(1500)
					.animate({'top':'300px'},500,'easeOutElastic').delay(1500)
					.animate({'left':'0px'},1000,'easeOutElastic').delay(1500)
					.animate({'top':'600px'},500,'easeOutElastic')
					/*連續動畫就直接.animate接下去即可*/
	})

＊jQ06片段:重複播放動畫

 

 

 

 

 

 

 

 

 

 

程式碼

	$(document).ready(function(){
		function boxer(){$('.box06').animate({'left':'400px'},1000,'easeOutElastic').delay(1000)
					.animate({'top':'1000px'},500,'easeOutElastic').delay(1000)
					.animate({'left':'0px'},1000,'easeOutElastic').delay(1000)
					.animate({'top':'200px'},500,'easeOutElastic')
					setTimeout(boxer,6000);}/*自己呼叫自己造成無窮迴圈*/
		setTimeout(boxer,100);/*先執行第一次函式*/			
	})
