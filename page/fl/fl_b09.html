移動左上的小方框 使右半邊可以看到更清楚的放大圖

程式碼


viewRec.buttonMode=true;
Lmap.buttonMode=true;
Lmap.mask=$mask;
var myRec1:Rectangle=new Rectangle(viewRec.x,viewRec.y,Smap.width-viewRec.width,Smap.height-viewRec.height);
var myRec2:Rectangle=new Rectangle($mask.width-Lmap.width+$mask.x,$mask.height-Lmap.height+$mask.y,Lmap.width-$mask.width,Lmap.height-$mask.height);

viewRec.addEventListener(MouseEvent.MOUSE_DOWN, $DragViewRec);
viewRec.addEventListener(MouseEvent.MOUSE_UP, $DragViewRec);
viewRec.addEventListener(MouseEvent.MOUSE_OUT, $DragViewRec);
viewRec.addEventListener(MouseEvent.MOUSE_MOVE, $DragViewRec);

Lmap.addEventListener(MouseEvent.MOUSE_DOWN, $DragLmap);
Lmap.addEventListener(MouseEvent.MOUSE_UP, $DragLmap);
Lmap.addEventListener(MouseEvent.MOUSE_OUT, $DragLmap);
Lmap.addEventListener(MouseEvent.MOUSE_MOVE, $DragLmap);


function $DragViewRec($e:MouseEvent){
	
	switch($e.type){
		
		case "mouseDown":
		viewRec.startDrag(false,myRec1);
		break;
		
		case "mouseUp":
		viewRec.stopDrag();
		break;
		
		case "mouseOut":
		viewRec.stopDrag();
		break;
		
		case "mouseMove":
		Lmap.x=$mask.x-(viewRec.x-Smap.x)*(Lmap.width-$mask.width)/(Smap.width-viewRec.width);
		Lmap.y=$mask.y-(viewRec.y-Smap.y)*(Lmap.height-$mask.height)/(Smap.height-viewRec.height);
		break;
	}
}
	
function $DragLmap($e:MouseEvent){
	
	switch($e.type){
		
		case "mouseDown":
		Lmap.startDrag(false,myRec2);
		break;
		
		case "mouseUp":
		Lmap.stopDrag();
		break;
		
		case "mouseOut":
		Lmap.stopDrag();
		break;
		
		case "mouseMove":
		viewRec.x=Smap.x-(Lmap.x-$mask.x)*(Smap.width-viewRec.width)/(Lmap.width-$mask.width);
		viewRec.y=Smap.y-(Lmap.y-$mask.y)*(Smap.height-viewRec.height)/(Lmap.height-$mask.height);
		break;
	}
}
