JQuery隨機動畫

＊jQ07片段:隨機動畫

 

 

 

 

 

程式碼

	$(document).ready(function(){
		function runBox(){ /*利用Math.random隨機產生0到1之間的數字*/
			/*宣告變數儲存隨機的X座標 將隨機產生數字乘上視窗寬度讓隨機的X座標不會超出視窗*/
            var randomX=Math.random()*$(window).width();
            /*隨機Y座標*/
			var randomY=Math.random()*$(window).height();
            /*隨機大小*/
			var randomSize=Math.random()*$(window).width()/2;
			/*隨機透明度*/
            var randomAlpha=Math.random();
            
			/*移動座標*/
			$('.box07').css('left',randomX+"px")
			.css('top',randomY+"px")
			.css('width','0%')
			.css('height','0%')
			.css('marginLeft','0px')
			.css('marginTop','0px')
			.css('opacity','0');
			
            /*顯現出大小跟透明度*/
			$('.box07').animate({
				'width':randomSize+"px",
				'height':randomSize+"px",
                /*由於box07的基準點是在左上角 有時會超出視窗
                  所以設定MarginLeft跟Margintop讓移動的基準點在box07中心*/
				'marginLeft':randomSize/2+"px",
				'marginTop':randomSize/2+"px",
				'opacity':randomAlpha,
				},1000);
			setTimeout(runBox,2500);
			}
		setTimeout(runBox,2500);	
	});
