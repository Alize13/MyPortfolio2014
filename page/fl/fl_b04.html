import flash.events.MouseEvent;

for(var i:uint=0;i< btn.numChildren;i++){
	trace(btn.getChildAt(i).name);
	var myBtn:MovieClip=btn.getChildAt(i) as MovieClip;
	myBtn.addEventListener(MouseEvent.MOUSE_OVER,$btnHandler);
	myBtn.addEventListener(MouseEvent.MOUSE_OUT,$btnHandler);
	myBtn.addEventListener(MouseEvent.CLICK,$btnHandler);
	myBtn.buttonMode=true;
	myBtn.mouseChildren=false;//不傳回子物件名稱(若用Target)  下面函式用currentTarget可以省略此行
}

function $btnHandler($e:MouseEvent){
	if($e.type=="mouseOver"){
	$e.currentTarget.gotoAndPlay("up");	
	//currentTarget目標層名稱 若用Target點到的元件名稱有可能是目標層子元件
	}
	
	if($e.type=="mouseOut"){
	$e.currentTarget.gotoAndPlay("down");
	}
	
	if($e.type=="click"){
		switch($e.target.name){
			case "first":
			this.pic_mc.gotoAndPlay("firstIn");
			break;
			case "second":
			this.pic_mc.gotoAndPlay("secondIn");
			break;
			case "third":
			this.pic_mc.gotoAndPlay("thirdIn");
			break;
		}
	}	
}
