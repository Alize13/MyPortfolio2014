var i=0;

function $add(){
	var _mc:MovieClip=new recMc();
	this.addChild(_mc);
	_mc.x = i % 8 * 80 + 40;
	_mc.y=Math.floor(i/8)*80+40;
	_mc.alpha = Math.random();
	
	i++;
	
	if(i==40){
		clearInterval(id);
		}
}

var id = setInterval($add,10);  //間隔10毫秒執行一次
